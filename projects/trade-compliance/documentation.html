Quick Start

// Initialize trade compliance analyzer
const analyzer = new TradeCompliance({
  country: 'italy',
  productCategory: 'electronics',
  updateFrequency: 'realtime'
});

// Calculate compliance costs

  shipmentValue: 50000,
  productType: 'consumer_goods'
});
